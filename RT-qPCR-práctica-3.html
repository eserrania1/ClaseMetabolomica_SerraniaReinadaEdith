Regresión lineal / Curva estándar

Primero obtener el valor de log de las concentraciones

Datos_curva_estándar <- Ct %>%
  filter(Muestra %in% c("50 ng", "10 ng", "5 ng", "1 ng", "0.5 ng", "0.1 ng")) %>%
  mutate(
    Concentracion = as.numeric(gsub(" ng", "", Muestra)),
    LogConc = log10(Concentracion)
  )

modelo_lineal <- lm(Ct ~ LogConc, data = Datos_curva)
coeficientes <- coef(modelo_lineal)

m <- round(coeficientes[2], 3)   # pendiente
b <- round(coeficientes[1], 3)   # intercepto
R2 <- round(summary(modelo_lineal)$r.squared, 4)

ecuacion_recta <- paste0("y = ", m, "x + ", b, "\nR2 = ", R2)
cat("y = ", m, "x + ", b, "\nR2 = ", R2)
